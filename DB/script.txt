CREATE SEQUENCE SEQ_DELIVERYPERSON_ID;

CREATE TABLE DeliveryPerson (
    ID INTEGER NOT NULL,
    Name VARCHAR(100) NOT NULL,
    Status VARCHAR(50) NOT NULL,
    PRIMARY KEY (ID)
);

SET TERM ^ ;

CREATE TRIGGER BI_DELIVERYPERSON FOR DeliveryPerson
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(SEQ_DELIVERYPERSON_ID, 1);
END^

SET TERM ; ^
